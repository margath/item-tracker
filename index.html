⚙️ Google Sheets Cloud Setup

To save data, this app connects to a Google Sheet using Google Apps Script.

  1. Create a new Google Sheet.
  2. At the bottom left, add a new tab and name it exactly: Players.
  3. In the Players tab, set up your roster starting in Row 2.
    Col A: Player Name | Col B: PIN | Col C: Character Name
  4. In the top menu, click Extensions > Apps Script.
  5. Delete any default code and paste the script below.
  6. Click Deploy > New deployment (or Manage Deployments > New Version).
  7. Select type Web app. Set 'Execute as' to Me and 'Who has access' to Anyone.
  8. Click Deploy, Authorize the permissions, and copy the generated Web app URL.
const SHEET_NAME = "LootData"; 
const PLAYER_SHEET = "Players"; 
const STANDARD_SHEET = "StandardItems";
const FOLDER_NAME = "Party_Loot_Images";

function getOrCreateFolder() { 
  const f = DriveApp.getFoldersByName(FOLDER_NAME); 
  return f.hasNext() ? f.next() : DriveApp.createFolder(FOLDER_NAME); 
}

function doGet(e) {
  try { 
    const ss = SpreadsheetApp.getActive();
    let s = ss.getSheetByName(SHEET_NAME); 
    let p = ss.getSheetByName(PLAYER_SHEET);
    let std = ss.getSheetByName(STANDARD_SHEET);
    
    let items = [];
    if (s) {
      let data = s.getDataRange().getValues();
      if (data.length === 1 && typeof data[0][0] === 'string' && data[0][0].trim().startsWith('[')) {
        items = JSON.parse(data[0][0]); 
      } else if (data.length > 1) {
        let headers = data[0];
        for (let i = 1; i < data.length; i++) {
          let item = {};
          headers.forEach((h, index) => { item[h] = data[i][index]; });
          items.push(item);
        }
      }
    }
    
    let players = [];
    if (p) {
      let data = p.getDataRange().getValues();
      for(let i=1; i < data.length; i++) { 
        if(data[i][0] && data[i][1]) {
          players.push({
            name: data[i][0].toString(), pin: data[i][1].toString(), character: data[i][2] ? data[i][2].toString() : ""
          });
        }
      }
    }

    let standardItems = [];
    if (std) {
      let data = std.getDataRange().getValues();
      if (data.length > 1) {
        let headers = data[0];
        for (let i = 1; i < data.length; i++) {
          let item = {};
          headers.forEach((h, index) => { item[h] = data[i][index]; });
          standardItems.push(item);
        }
      }
    }
    return ContentService.createTextOutput(JSON.stringify({items: items, players: players, standardItems: standardItems})).setMimeType(ContentService.MimeType.JSON);
  } catch (err) { 
    return ContentService.createTextOutput(JSON.stringify({items: [], players: [], standardItems: []})).setMimeType(ContentService.MimeType.JSON); 
  }
}

function doPost(e) {
  try { 
    const p = JSON.parse(e.postData.contents);
    if (p.action === 'uploadImage') {
      const f = getOrCreateFolder(), b = Utilities.newBlob(Utilities.base64Decode(p.base64), p.mimeType, p.fileName);
      const file = f.createFile(b); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
      return ContentService.createTextOutput(JSON.stringify({ url: "https://drive.google.com/uc?id=" + file.getId() })).setMimeType(ContentService.MimeType.JSON);
    }
    if (p.action === 'saveData') {
      let s = SpreadsheetApp.getActive().getSheetByName(SHEET_NAME); 
      if (!s) s = SpreadsheetApp.getActive().insertSheet(SHEET_NAME);
      s.clear(); 
      if (p.data && p.data.length > 0) {
        let headers = ["id", "apparentName", "trueName", "imageUrl", "owner", "weight", "quantity", "isConsumable", "isStandard", "isEquipped", "requiresAttunement", "isAttuned", "isIdentified", "description", "isCurseRevealed", "curseDetectDC", "curseDetails", "secretKeeper"];
        let rows = [headers];
        p.data.forEach(item => { 
          let rowData = headers.map(h => {
            let val = item[h];
            if (val === undefined || val === null) return "";
            if (typeof val === 'boolean') return val ? "TRUE" : "FALSE";
            return val;
          });
          rows.push(rowData); 
        });
        s.getRange(1, 1, rows.length, headers.length).setValues(rows);
      }

      let std = SpreadsheetApp.getActive().getSheetByName(STANDARD_SHEET); 
      if (!std) std = SpreadsheetApp.getActive().insertSheet(STANDARD_SHEET);
      std.clear(); 
      if (p.standardData && p.standardData.length > 0) {
        let stdHeaders = ["trueName", "apparentName", "description", "weight", "isConsumable"];
        let stdRows = [stdHeaders];
        p.standardData.forEach(item => {
          let rowData = stdHeaders.map(h => {
            let val = item[h];
            return typeof val === 'boolean' ? (val ? "TRUE" : "FALSE") : (val || "");
          });
          stdRows.push(rowData);
        });
        std.getRange(1, 1, stdRows.length, stdHeaders.length).setValues(stdRows);
      }
      return ContentService.createTextOutput(JSON.stringify({ status: "success" })).setMimeType(ContentService.MimeType.JSON);
    }
  } catch (err) { 
    return ContentService.createTextOutput(JSON.stringify({ error: err.toString() })).setMimeType(ContentService.MimeType.JSON); 
  }
}
